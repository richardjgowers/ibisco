6 ! If SX is n times (where n is even) greater than half side of the box
7 ! (in the x-direction in this case) SPX represent the "percentage" of the
8 ! position in the half part of the box where is located the image of the
9 ! atom. Therefore the image is located at (SPX * BOXX2), because if n is even
10 ! INT(SPX) is always 0.
11 !
12 ! If SX is n times (where n is odd) greater than half side of the box
13 ! (in the x-direction in this case) (SPX - 2.0*INT(SPX)) represent the
14 ! "percentage" of the position in the half part of the box where is located
15 ! the image of the atom. If n is odd [SPX > 1].
16 !
17 !
18 !
19 !
20 SUBROUTINE shift()
21 
22  USE var
23  USE omp_lib
24  IMPLICIT NONE
25  INTEGER :: i
26  REAL*8 ::spx, spy, spz
27  ! *******************************************************************
28 
29  ! INT(funzione parte intera) is an intrinsic function that reads a real, cuts the decimal
30  ! part of the number leaving only its integer part (the final format is real)
31 
32  !$OMP PARALLEL DO SCHEDULE(STATIC,1) NUM_THREADS(7) DEFAULT(NONE) &
33  !$OMP& SHARED(NATOMS,SX,SY,SZ,BOXX2,BOXY2,BOXZ2,RX,RY,RZ)&
34  !$OMP& private(I,SPX,SPY,SPZ)
35  DO i = 1, natoms
36  IF ((sx(i)> boxx2).OR.(sx(i) < -boxx2).OR.(sy(i)> boxy2) &
37  .OR.(sy(i)< -boxy2).OR.(sz(i)> boxz2).OR.(sz(i)<-boxz2)) THEN
38 
39  rx(i) = sx(i)/boxx2
40  ry(i) = sy(i)/boxy2
41  rz(i) = sz(i)/boxz2
42 
43  spx = rx(i) - 2.0*int(rx(i)/2.0)
44  spy = ry(i) - 2.0*int(ry(i)/2.0)
45  spz = rz(i) - 2.0*int(rz(i)/2.0)
46 
47  rx(i) = (spx - 2.0*int(spx))*boxx2
48  ry(i) = (spy - 2.0*int(spy))*boxy2
49  rz(i) = (spz - 2.0*int(spz))*boxz2
50  ELSE
51 
52  rx(i) = sx(i)
53  ry(i) = sy(i)
54  rz(i) = sz(i)
55  END IF
56  END DO
57  !$OMP END PARALLEL DO
58 
59  RETURN
60 END SUBROUTINE shift
