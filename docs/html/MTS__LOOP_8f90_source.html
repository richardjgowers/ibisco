71 SUBROUTINE mts_force(STEPNO)

82  fx = 0.0d0
83  fy = 0.0d0
84  fz = 0.0d0
85 
86  fxnb = 0.0d0
87  fynb = 0.0d0
88  fznb = 0.0d0

90  pt11 = 0.0d0
91  pt22 = 0.0d0
92  pt33 = 0.0d0
93  pt12 = 0.0d0
94  pt13 = 0.0d0
95  pt23 = 0.0d0
96 
97  !Reset potential energy measures
98  v_nb = 0.0
99  v_bond = 0.0
100  v_angle = 0.0
101  v_torsion = 0.0
102  v_oop = 0.0
103 
104  mts_mod = mod(stepno, (3 + nmts))
105  SELECT CASE(mts_mod)
106  CASE(1, 2, 3) ! Explicit steps
107  CALL nonbonded_force(ncoarse, bead, maxnab_bead, list_bead, rcut_bead, rcutsq_bead)
108 
109  CALL mts_saveforce(mts_mod, bead, fx, fy, fz, nitems, mts_fx, mts_fy, mts_fz, ncoarse)
110  CALL mts_saveauxs(mts_mod)
111 
112  CASE default ! Use approximation
113  CALL mts_approx(mts_mod, bead, fx, fy, fz, nitems, mts_fx, mts_fy, mts_fz, ncoarse)
114  CALL mts_loadauxs(mts_mod)
115  END SELECT
116  CALL distribute_vsforce()
117 
118  ! Call atoms second, so that V_NB and PTxx only have CG contributions
119  ! Order shouldnt matter anyway...
120  CALL nonbonded_force(numatoms, atom, maxnab_atom, list_atom, rcut_atom, rcutsq_atom)
121 
122  DO i = 1, natoms
123  fxnb(i) = fx(i)
124  fynb(i) = fy(i)
125  fznb(i) = fz(i)
126  END DO
127 
128  CALL bonded_force()
129 
130  CALL distribute_vsforce()
131 
132  DO i=1,natoms
133  pt11 = pt11 + (fx(i) - fxnb(i))*sx(i)
134  pt22 = pt22 + (fy(i) - fynb(i))*sy(i)
135  pt33 = pt33 + (fz(i) - fznb(i))*sz(i)
136  pt12 = pt12 + (fy(i) - fynb(i))*sx(i)
137  pt13 = pt13 + (fz(i) - fznb(i))*sx(i)
138  pt23 = pt23 + (fz(i) - fznb(i))*sy(i)
139  END DO
140 
141  RETURN
